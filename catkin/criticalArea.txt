for(int i = 0; i < path.poses.size()-1; i++){
		//Iterate through poses minus last because it is the goal pose and has no next!
		path.poses.at(i).pose.orientation.x = (path.poses.at(i+1).point.x - path.poses.at(i).pose.point.x) * (path.poses.at(i+1).point.y - path.poses.at(i).pose.point.y);
		path.poses.at(i).pose.orientation.y = (path.poses.at(i+1).point.x - path.poses.at(i).pose.point.x) * (path.poses.at(i+1).point.y - path.poses.at(i).pose.point.y);
		path.poses.at(i).pose.orientation.z = (path.poses.at(i+1).point.x - path.poses.at(i).pose.point.x) * (path.poses.at(i+1).point.y - path.poses.at(i).pose.point.y);
		path.poses.at(i).pose.orientation.w = math::sqrt((math::pow((path.poses.at(i+1).point.x - path.poses.at(i).pose.point.x), 2) + math::pow((path.poses.at(i+1).point.y - path.poses.at(i).pose.point.y), 2)));
	}
